# Ba-Ge — Python deps. The UI is PySide6 (Qt): its wheels are fully
# self-contained (no system Tk/GTK), and QSystemTrayIcon speaks StatusNotifier
# natively, so the tray works on GNOME without PyGObject. One codebase serves
# Linux, macOS, and Windows. Install into a uv-managed standalone Python:
#   uv venv --python 3.12 .venv && uv pip install -r requirements.txt
#
# System tools still used on Linux (not Python packages):
#   alsa-utils    microphone capture (arecord)
#   ffmpeg        file transcription
#   libnotify-bin notifications (notify-send)
# Paste injection needs no system tool: the keystroke is sent with pynput (XTEST,
# X11) and the clipboard is handled by Qt — both bundled. libxcb-cursor0 is what
# Qt's xcb platform plugin needs on some minimal systems.

PySide6-Essentials>=6.7    # Qt UI toolkit + tray (all OSes; self-contained wheel)
pynput>=1.7                # global hotkey (all OSes) + paste keystroke/typing (also pulls python-xlib on Linux)
platformdirs>=4            # cross-platform config/cache locations
pyperclip>=1.8             # clipboard (paste injection on macOS/Windows)
sounddevice>=0.4           # audio capture on macOS/Windows (Linux uses arecord)
# imageio-ffmpeg           # optional: bundled ffmpeg for file transcription on macOS/Windows
